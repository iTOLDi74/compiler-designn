%{
    #include <stdio.h>
    #include "y.tab.h"  
    #include <stdlib.h>
    extern int yylval;
%}

%%

[0-9]+      { yylval = atoi(yytext); return NUMBER; }
"+"         { return PLUS; }
"-"         { return MINUS; }
"*"         { return TIMES; }
"/"|"รท"     { return DIVIDE; }
"%"         { return MOD; }
"("         { return LPAREN; }  
")"         { return RPAREN; }  
\n          { return EOL; }
[ \t\r]     {}  
.           { printf("Invalid char: %s\n", yytext); }  

%%

int yywrap(void) { return 1; }
